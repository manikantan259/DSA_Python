def twoSum(arr , target):
    
    left , right = 0 , len(arr)-1
    
    while left < right :
        
        calSum = arr[left] + arr[right] 
        print (calSum)
        if calSum ==  target :
            return [left , right]
            
        elif calSum < target:
            left +=1
        else :
            right -=1
            
    return []
    
arr = [2,7,11,15]   
print (twoSum(arr , 9))

def twoSum(arr , target):
    
    dictMap = {}
    
    for i in range (0 , len(arr)):
        diff = target - arr[i]
        
        if diff in dictMap:
            return [dictMap[diff] , i]
        dictMap [arr[i]] = i
        
    return []
